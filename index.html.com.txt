        // Logic to get User Name
        window.onload = function() {
            let userName = localStorage.getItem('dmind_user');
            
            if (!userName) {
                userName = prompt("Welcome to D-Mind! Please enter your name:");
                if (userName) {
                    localStorage.setItem('dmind_user', userName);
                } else {
                    userName = "Guest";
                }
            }
            
            document.getElementById('greetingText').innerText = `Hello, ${userName}. What's the plan today?`;
        };

        // Search Trigger
        document.getElementById('mainSearch').addEventListener('keypress', function (e) {
            if (e.key === 'Enter') {
                const query = this.value;
                alert('D-Mind is processing: ' + query);
            }
        });
